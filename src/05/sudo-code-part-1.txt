
const p = ['c', 'a', 'b', 'B', 'A', 'D'];

let restart = true;

while (restart) {

  let found = false;
  for (for i in p) {

    // check that p[i] is lower case 
    // and check that p[i+1] is uppercase
    if (p[i] is lower case && p[i+1] is uppercase) {

      // check if chars react
      if (p[i] === p[i + 1]) {
        // remove p[i] and p[i+1] from p
        found = true
        break;
      }
    } 
  }

  if (!found) {
    restart = false;
  }
}

return p.length
